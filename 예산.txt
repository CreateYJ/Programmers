import java.util.Arrays;
class Solution {
    public int solution(int[] d, int budget) {
        int answer = 0;
        int mon=0;;
        Arrays.sort(d);
        for(int i = 0; i<d.length; i++) {
            mon+=d[i];
            if(mon<=budget)
                answer=i+1;
            else
                break;
        }
        return answer;
    }
}
